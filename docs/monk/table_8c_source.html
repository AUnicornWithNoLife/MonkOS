73 /// Re-sort all unsorted region records starting from the requested record.
74 static void
75 resort(memregion_t *r, memregion_t *term)
76 {
77  while (r + 1 < term) {
78  if (cmp_region(r, r + 1) < 0)
79  break;
80  memregion_t tmp = r[0];
81  r[0] = r[1];
82  r[1] = tmp;

87 /// Find all overlapping memory regions in the memory table and collapse or
88 /// reorganize them.
89 static void
90 collapse_overlaps()
91 {
92  memregion_t *curr = table->region;
93  memregion_t *term = table->region + table->count;
94 
95  while (curr + 1 < term) {
96 
97  // Collapse empty entries.
98  if (curr->size == 0) {
99  collapse(curr, term--);
100  continue;
101  }
102 
103  memregion_t *next = curr + 1;
104 
105  uint64_t cl = curr->addr;
106  uint64_t cr = curr->addr + curr->size;
107  uint64_t nl = next->addr;
108  uint64_t nr = next->addr + next->size;
109 
110  // No overlap? Then go to the next region.
111  if (min(cr, nr) <= max(cl, nl)) {
112  curr++;
113  continue;
114  }
115 
116  // Handle the 5 alignment cases:
117  // xxx xxx xxxx xxx xxxxx
118  // yyy yyyy yyy yyy yyy
119  // The remaining cases are impossible due to sorting.
120 
121  if (cl == nl) {
122  if (cr == nr) {
123  if (next->type > curr->type) {
124  // 111 -> 222
125  // 222
126  collapse(curr, term--);
127  }
128  else {
129  // 222 -> 222
130  // 111
131  collapse(next, term--);
132  }
133  }
134  else { /* if cr != nr */
135  if (next->type > curr->type) {
136  // 111 -> 2222
137  // 2222
138  collapse(curr, term--);
139  }
140  else {
141  // 222 -> 222
142  // 1111 -> 1
143  next->size = nr - cr;
144  next->addr = cr;
145  resort(next, term);
146  }
147  }
148  }
149 
150  else { /* if cl != nl */
151  if (cr == nr) {
152  if (next->type > curr->type) {
153  // 1111 -> 1
154  // 222 -> 222
155  curr->size = nl - cl;
156  }
157  else {
158  // 2222 -> 2222
159  // 111
160  collapse(next, term--);
161  }
162  }
163  else if (cr < nr) {
164  if (next->type > curr->type) {
165  // 1111 -> 1
166  // 2222 -> 2222
167  curr->size = nl - cl;
168  }
169  else {
170  // 2222 -> 2222
171  // 1111 -> 1
172  next->size = nr - cr;
173  next->addr = cr;
174  resort(next, term);
175  }
176  }
177  else { /* if cr > nr */
178  if (next->type > curr->type) {
179  // 11111 -> 1
180  // 222 -> 222
181  // -> 1
182  curr->size = nl - cl;
183  insertafter(next, term++);
184  next[1].addr = nr;
185  next[1].size = cr - nr;
186  next[1].type = curr->type;
187  next[1].flags = curr->flags;
188  resort(next + 1, term);
189  }
190  else {
191  // 22222 -> 22222
192  // 111
193  collapse(next, term--);
194  }
195  }
196  }

201 /// Find missing memory regions in the table, and fill them with entries of
202 /// the requested type.
203 static void
204 fill_gaps(int32_t type)
205 {
206  memregion_t *curr = table->region;
207  memregion_t *term = table->region + table->count;
208 
209  while (curr + 1 < term) {
210  memregion_t *next = curr + 1;
211 
212  uint64_t cr = curr->addr + curr->size;
213  uint64_t nl = next->addr;
214 
215  if (cr == nl) {
216  curr++;
217  continue;
218  }
219 
220  // Try to expand one of the neighboring entries if one of them has the
221  // same type as the fill type.
222  if (curr->type == type) {
223  curr->size += nl - cr;
224  }
225  else if (next->type == type) {
226  next->size += nl - cr;
227  next->addr = cr;
228  }
229 
230  // Neither neighboring region has the fill type, so insert a new
231  // region record.
232  else {
233  insertafter(curr, term++);
234  next->addr = cr;
235  next->size = nl - cr;
236  next->type = type;
237  next->flags = 0;
238  }
239  }
240 }

274 static void
275 normalize()
276 {
277  // Sort the memory table by address.
278  qsort(table->region, table->count, sizeof(memregion_t),
279  cmp_region);
280 
281  // Remove overlapping regions.

285  fill_gaps(MEMTYPE_RESERVED);
