71 /// Reserve an aligned region of memory managed by the memory table module.
72 static void *
73 reserve_region(const memtable_t *table, uint64_t size, uint32_t alignshift)
74 {
75  const memregion_t *r = table->region;
76  const memregion_t *t = table->region + table->count;
77  for (; r < t; r++) {
78 
79  // Skip reserved regions and regions that are too small.
80  if (r->type != MEMTYPE_USABLE)
81  continue;
82  if (r->size < size)
83  continue;
84 
85  // Find address of first properly aligned byte in the region.
86  uint64_t paddr = r->addr + (1 << alignshift) - 1;
87  paddr >>= alignshift;
88  paddr <<= alignshift;
89 
90  // If the region is too small after alignment, skip it.
91  if (paddr + size > r->addr + r->size)
92  continue;
93 
94  // Reserve the aligned memory region and return its address.
95  memtable_add(paddr, size, MEMTYPE_RESERVED);
96  return (void *)paddr;
97  }
98  return NULL;
99 }

101 void
102 page_init()
103 {
104  const memtable_t *table = memtable();
105  if (table->last_usable == 0)
106  fatal();
107 
108  // Calculate the size of the page frame database. It needs to describe
109  // each page up to and including the last physical address.
110  pgcontext.pfcount = table->last_usable / PAGE_SIZE;
111  uint64_t pfdbsize = pgcontext.pfcount * sizeof(pf_t);
112 
113  // Round the database size up to the nearest 2MiB since we'll be using
114  // large pages in the kernel page table to describe the database.
115  pfdbsize += PAGE_SIZE_LARGE - 1;
116  pfdbsize >>= PAGE_SHIFT_LARGE;
117  pfdbsize <<= PAGE_SHIFT_LARGE;
118 
119  // Find a contiguous, 2MiB-aligned region of memory large enough to hold
120  // the entire pagedb.
121  pgcontext.pfdb =
122  (pf_t *)reserve_region(table, pfdbsize, PAGE_SHIFT_LARGE);
123  if (pgcontext.pfdb == NULL)
124  fatal();
125 
126  // Identity-map all physical memory into the kernel's page table.
127  map_memory();
128 
129  // Keep track of the root PML4 table.
130  pgcontext.pagetable = (page_t *)MEM_KERNEL_PAGETABLE;
131 
132  // Create the page frame database in the newly mapped virtual memory.
133  memzero(pgcontext.pfdb, pfdbsize);
134 
135  // Initialize available page frame list.
136  pgcontext.pfavail = 0;
137  pgcontext.pfnhead = PFN_INVALID;
138  pgcontext.pfntail = PFN_INVALID;
139 
140  // Traverse the memory table, adding page frame database entries for each
141  // region in the table.
142  for (uint64_t r = 0; r < table->count; r++) {
143  const memregion_t *region = &table->region[r];
144 
145  // Ignore non-usable memory regions.
146  if (region->type != MEMTYPE_USABLE)
147  continue;
148 
149  // Create a chain of available page frames.
150  uint64_t pfn0 = region->addr >> PAGE_SHIFT;
151  uint64_t pfnN = (region->addr + region->size) >> PAGE_SHIFT;
152  for (uint64_t pfn = pfn0; pfn < pfnN; pfn++) {
153  pf_t *pf = PFN_TO_PF(pfn);
154  pf->prev = pfn - 1;
155  pf->next = pfn + 1;
156  pf->type = PFTYPE_AVAILABLE;
157  }
158 
159  // Link the chain to the list of available frames.
160  if (pgcontext.pfntail == PFN_INVALID)
161  pgcontext.pfnhead = pfn0;
162  else
163  pgcontext.pfdb[pgcontext.pfntail].next = pfn0;
164  pgcontext.pfdb[pfn0].prev = pgcontext.pfntail;
165  pgcontext.pfdb[pfnN - 1].next = PFN_INVALID;
166  pgcontext.pfntail = pfnN - 1;
167 
168  // Update the total number of available frames.
169  pgcontext.pfavail += (uint32_t)(pfnN - pfn0);
170  }
171 
172  // Install page fault handler
173 }

175 static pf_t *
176 pfalloc()
177 {
178  // For now, fatal out. Later, we'll add swapping.
179  if (pgcontext.pfavail == 0)
180  fatal();
181 
182  // Grab the first available page frame from the database.
183  pf_t *pfdb = pgcontext.pfdb;
184  pf_t *pf = PFN_TO_PF(pgcontext.pfnhead);
185 
186  // Update the available frame list.
187  pgcontext.pfnhead = pfdb[pgcontext.pfnhead].next;
188  if (pgcontext.pfnhead != PFN_INVALID)
189  pfdb[pgcontext.pfnhead].prev = PFN_INVALID;
190 
191  // Initialize and return the page frame.
192  memzero(pf, sizeof(pf_t));
193  pf->refcount = 1;
194  pf->type = PFTYPE_ALLOCATED;
195  return pf;
196 }

198 static void
199 pffree(pf_t *pf)
200 {
201  if (pf->type != PFTYPE_ALLOCATED)
202  fatal();
203 
204  pf_t *pfdb = pgcontext.pfdb;
205 
206  // Re-initialize the page frame record.
207  memzero(pf, sizeof(pf_t));
208  pf->prev = PFN_INVALID;
209  pf->next = pgcontext.pfnhead;
210  pf->type = PFTYPE_AVAILABLE;
211 
212  // Update the database's available frame list.
213  uint32_t pfn = PF_TO_PFN(pf);
214  pfdb[pgcontext.pfnhead].prev = pfn;
215  pgcontext.pfnhead = pfn;
216 
217  pgcontext.pfavail++;
218 }

220 static uint64_t
221 pgalloc()
222 {
223  // Allocate a page frame from the db and compute its physical address.
224  pf_t *pf = pfalloc();
225  uint64_t paddr = PF_TO_PADDR(pf);
226 
227  // Always zero the contents of newly allocated pages.
228  memzero((void *)paddr, PAGE_SIZE);
229 
230  // Return the page's physical address.
231  return paddr;
232 }

242 static void
243 pgfree_recurse(page_t *page, int level)
244 {
245  // If we're at the PT level, return the leaf pages to the page frame
246  // database.
247  if (level == 1) {
248  for (uint64_t e = 0; e < 512; e++) {
249  uint64_t paddr = PTE_TO_PADDR(page->entry[e]);
250  if (paddr == 0)
251  continue;
252 
253  pf_t *pf = PADDR_TO_PF(paddr);
254  if (pf->type == PFTYPE_ALLOCATED)
255  pgfree(paddr);
256  }
257  }
258 
259  // If we're at the PD level or above, recursively traverse child tables
260  // until we reach the PT level.
261  else {
262  for (uint64_t e = 0; e < 512; e++) {
263  page_t *child = PGPTR(page->entry[e]);
264  if (child != NULL)
265  pgfree_recurse(child, level - 1);
266  }
267  }
268 }

270 /// Add to the page table an entry mapping a virtual address to a physical
271 /// address.
272 static void
273 add_pte(pagetable_t *pt, uint64_t vaddr, uint64_t paddr, uint32_t pflags)
274 {
275  // Keep track of the pages we add to the page table as we add this new
276  // page table entry.
277  uint64_t added[3];
278  int count = 0;
279 
280  // Decompose the virtual address into its hierarchical table components.
281  uint32_t pml4e = PML4E(vaddr);
282  uint32_t pdpte = PDPTE(vaddr);
283  uint32_t pde = PDE(vaddr);
284  uint32_t pte = PTE(vaddr);
285 
286  // Follow the page table hierarchy down to the lowest level, creating
287  // new pages for tables as needed.
288 
289  page_t *pml4t = pt->pml4t;
290  if (pml4t->entry[pml4e] == 0) {
291  uint64_t pgaddr = pgalloc();
292  added[count++] = pgaddr;
293  pml4t->entry[pml4e] = pgaddr | PF_PRESENT | PF_RW;
294  }
295 
296  page_t *pdpt = PGPTR(pml4t->entry[pml4e]);
297  if (pdpt->entry[pdpte] == 0) {
298  uint64_t pgaddr = pgalloc();
299  added[count++] = pgaddr;
300  pdpt->entry[pdpte] = pgaddr | PF_PRESENT | PF_RW;
301  }
302 
303  page_t *pdt = PGPTR(pdpt->entry[pdpte]);
304  if (pdt->entry[pde] == 0) {
305  uint64_t pgaddr = pgalloc();
306  added[count++] = pgaddr;
307  pdt->entry[pde] = pgaddr | PF_PRESENT | PF_RW;
308  }
309 
310  // Add the page table entry.
311  page_t *ptt = PGPTR(pdt->entry[pde]);
312  ptt->entry[pte] = paddr | pflags;
313 
314  // If adding the new entry required the page table to grow, make sure to
315  // add the page table's new pages as well.
316  for (int i = 0; i < count; i++)
317  add_pte(pt, (uint64_t)pt->next++, added[i], PF_PRESENT | PF_RW);
318 }

320 static uint64_t
321 remove_pte(pagetable_t *pt, uint64_t vaddr)
322 {
323  // Decompose the virtual address into its hierarchical table components.
324  uint32_t pml4e = PML4E(vaddr);
325  uint32_t pdpte = PDPTE(vaddr);
326  uint32_t pde = PDE(vaddr);
327  uint32_t pte = PTE(vaddr);
328 
329  // Traverse the hierarchy, looking for the page.
330  page_t *pml4t = pt->pml4t;
331  page_t *pdpt = PGPTR(pml4t->entry[pml4e]);
332  page_t *pdt = PGPTR(pdpt->entry[pdpte]);
333  page_t *ptt = PGPTR(pdt->entry[pde]);
334  page_t *pg = PGPTR(ptt->entry[pte]);
335 
336  // Clear the page table entry for the virtual address.
337  ptt->entry[pte] = 0;
338 
339  // Return the physical address of the page table entry that was removed.
340  return (uint64_t)pg;
341 }

344 pagetable_create(pagetable_t *pt, void *vaddr)
345 {
