121  tty_printf(TTY_CONSOLE, "Local APIC id %u: %s\n",

180 static bool
181 command_exec(const char *cmd)
182 {
183  if (cmd[0] == 0)
184  return true;
185 
186  for (int i = 0; i < arrsize(commands); i++) {
187  if (!strcmp(commands[i].str, cmd))
188  return commands[i].run();
189  }
190 
191  tty_printf(TTY_CONSOLE, "Unknown command: %s\n", cmd);
192  return true;
193 }

201 static void
202 command_run()
203 {
204  char cmd[256];
205  int cmdlen = 0;
206 
207  for (;;) {
208  halt();
209 
210  key_t key;
211  bool avail;
212  while ((avail = kb_getkey(&key)) != false) {
213 
214  // If a printable character was typed, append it to the command.
215  if (key.ch >= 32 && key.ch < 127) {
216  if (cmdlen < arrsize(cmd) - 1) {
217  cmd[cmdlen] = key.ch;
218  tty_printc(TTY_CONSOLE, cmd[cmdlen]);
219  cmdlen++;
220  }
221  }
222 
223  // Handle special keys (like enter, backspace).
224  else if (key.brk == KEYBRK_DOWN) {
225 
226  if (key.code == KEY_ENTER) {
227  tty_printc(TTY_CONSOLE, '\n');
228 
229  // Strip trailing whitespace.
230  while (cmdlen > 0 && cmd[cmdlen - 1] == ' ')
231  cmdlen--;
232  cmd[cmdlen] = 0;
233 
234  // Execute the command.
235  bool cont = command_exec(cmd);
236  cmdlen = 0;
237  if (cont)
238  command_prompt();
239  else
240  return;
241  }
242 
243  else if (key.code == KEY_BACKSPACE && cmdlen > 0) {
244  tty_printc(TTY_CONSOLE, '\b');
245  cmdlen--;
246  }
247 
248  }
249  }
250  }
251 }

253 static void
254 keycode_run()
255 {
256  for (;;) {
257  halt();
258 
259  key_t key;
260  bool avail;
261  while ((avail = kb_getkey(&key)) != false) {
262  if (key.ch) {
263  tty_printf(
264  TTY_CONSOLE,
265  "Keycode: \033[%c]%02x\033[-] meta=%02x '%c'\n",
266  key.brk == KEYBRK_UP ? 'e' : '2',
267  key.code,
268  key.meta,
269  key.ch);
270  }
271  else {
272  tty_printf(
273  TTY_CONSOLE,
274  "Keycode: \033[%c]%02x\033[-] meta=%02x\n",
275  key.brk == KEYBRK_UP ? 'e' : '2',
276  key.code,
277  key.meta);
278  }
279  if ((key.brk == KEYBRK_UP) && (key.meta & META_ALT) &&
280  (key.code == KEY_TAB)) {
281  switch_mode(&mode_command);
282  return;
283  }
284  }
285  }
