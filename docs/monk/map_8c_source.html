132 static void
133 map_region(const memtable_t *table, const memregion_t *region)
134 {
135  // Don't map bad (or unmapped) memory.
136  if (region->type == MEMTYPE_UNMAPPED || region->type == MEMTYPE_BAD)
137  return;
138 
139  // Don't map reserved regions beyond the last usable physical address.
140  if (region->type == MEMTYPE_RESERVED &&
141  region->addr >= table->last_usable)
142  return;
143 
144  uint64_t addr = region->addr;
145  uint64_t term = region->addr + region->size;
146 
147  // Create a series of pages that cover the region. Try to use the largest
148  // page sizes possible to keep the page table small.
149  while (addr < term) {
150  uint64_t remain = term - addr;
151 
152  // Create a huge page (1GiB) if possible.
153  if ((addr & (PAGE_SIZE_HUGE - 1)) == 0 &&
154  (remain >= PAGE_SIZE_HUGE)) {
155  create_huge_page(addr, region->type);
156  addr += PAGE_SIZE_HUGE;
157  }
158 
159  // Create a large page (2MiB) if possible.
160  else if ((addr & (PAGE_SIZE_LARGE - 1)) == 0 &&
161  (remain >= PAGE_SIZE_LARGE)) {
162  create_large_page(addr, region->type);
163  addr += PAGE_SIZE_LARGE;
164  }
165 
166  // Create a small page (4KiB).
167  else {
168  create_small_page(addr, region->type);
169  addr += PAGE_SIZE;
170  }
171  }
172 }

174 void
175 map_memory()
176 {
177  // Zero all page table memory.
178  memzero((void *)MEM_KERNEL_PAGETABLE, MEM_KERNEL_PAGETABLE_SIZE);
179 
180  kptable.root = (page_t *)MEM_KERNEL_PAGETABLE;
181  kptable.next_page = (page_t *)(MEM_KERNEL_PAGETABLE + PAGE_SIZE);
182  kptable.term_page = (page_t *)MEM_KERNEL_PAGETABLE_END;
183 
184  // For each memory region in the BIOS-supplied memory table, create
185  // appropriate page table entries.
186  const memtable_t *table = memtable();
187  for (uint64_t r = 0; r < table->count; r++)
188  map_region(table, &table->region[r]);
189 
190  // Install the new page table.
191  uint64_t addr = (uint64_t)kptable.root;
192  set_pagetable(addr);
193 }
